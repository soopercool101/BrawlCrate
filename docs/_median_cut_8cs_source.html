528 
529  //Reset bounds and weight
530  _min = 0xFFFFFFFF;
531  _max = 0;
532  _weight = 0;
533 
534  //Zero errors (for use as color accumulator)
535  aError = rError = gError = bError = 0;
536  //Memory.Fill(&box->error, 32, 0);
537 
538  //Get min/max bounds for all contained color entries and calculate color
539  for (current = _rootEntry->_next; current != _rootEntry; current = current->_next)
540  {
541  _weight += weight = current->_weight;
542  tPtr = (byte*) &current->_color;
543 
544  //Update bounds for and accumulate each element
545  for (int i = 0; i < 4; i++)
546  {
547  val = *tPtr++;
548  pMin[i] = Math.Min(val, pMin[i]);
549  pMax[i] = Math.Max(val, pMax[i]);
550  colBuffer[i] += val * weight;
551  }
552  }
553 
554  //Set calculated color
555  sColor = (byte*) &box->_color;
556  for (int i = 0; i < 4; i++)
557  {
558  sColor[i] = (byte) (colBuffer[i] / _weight);
559  }
560 
561  //Calculate volume
562  _volume = 1;
563  for (int i = 0; i < 4; i++)
564  {
565  diff = pMax[i] - pMin[i] + 1;
566  _volume *= (uint) diff;
567  size[i] = (byte) diff;
568  }
569 
570  if (_volume == 0)
571  {
572  _volume = 0xFFFFFFFF;
573  }
574 
575  //Calculate error
576  aError = rError = gError = bError = 0;
577  //Memory.Fill(&box->error, 32, 0);
578  for (current = _rootEntry->_next; current != _rootEntry; current = current->_next)
579  {
580  weight = current->_weight;
581  tPtr = (byte*) &current->_color;
582  for (int i = 0; i < 4; i++)
583  {
584  diff = tPtr[i] - sColor[i];

586  }
587  }
588

591  for (int i = 0; i < 4; i++)
