65  public void SetCamWithBox(Vector3 min, Vector3 max)
66  {
67  GLCamera cam = CurrentViewport.Camera;
68 
69  //Get the position of the midpoint of the bounding box plane closer to the camera
70  Vector3 frontMidPt = new Vector3((max._x + min._x) / 2.0f, (max._y + min._y) / 2.0f, max._z);
71  float tan = (float)Math.Tan(cam.VerticalFieldOfView / 2.0f * Maths._deg2radf), distX = 0, distY = 0;
72 
73  //The tangent value would only be 0 if the FOV was 0,
74  //meaning nothing would be visible anyway
75  if (tan != 0)
76  {
77  //Calculate lengths
78  Vector3 extents = max - min;
79  Vector3 halfExtents = extents / 2.0f;
80  if (halfExtents._y != 0.0f)
81  {
82  float ratio = halfExtents._x / halfExtents._y;
83  distY = halfExtents._y / tan; //The camera's distance from the model's midpoint in respect to Y
84  distX = distY * ratio;
85  }
86  else if (halfExtents._x != 0.0f)
87  {
88  float ratio = halfExtents._y / halfExtents._x;
89  distX = halfExtents._x / tan; //The camera's distance from the model's midpoint in respect to Y
90  distY = distX * ratio;
91  }
92  }
93 
94  cam.Reset();
95 
96  cam.Translate(frontMidPt._x, frontMidPt._y, Maths.Max(distX, distY, max._z) + 2.0f);
97  if (CurrentViewport.GetProjectionType() == ViewportProjection.Orthographic)
98  {
99  cam.Zoom(Maths.Max(distX, distY, max._z) / (Maths.Max(Size.Height, Size.Width) * 0.01f));
100  }
101 
102  Invalidate();
103  }
