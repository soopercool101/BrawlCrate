339  private void lstColors_DoubleClick(object sender, EventArgs e)
340  {
341  IList<int> indices = SelectedIndices;
342  if (_colorSource == null || indices.Count <= 0)
343  {
344  return;
345  }
346 
347  int count = indices.Count;
348  if (count == 1)
349  {
350  tempIndex = indices[0];
351  ARGBPixel prev = (ARGBPixel) lstColors.Items[tempIndex];
352  _dlgColor.Color = (Color) prev;
353  _dlgColor.OnColorChanged += _dlgColor_OnColorChanged;
354  if (_dlgColor.ShowDialog(this) != DialogResult.OK)
355  {
356  if (tempIndex >= 0)
357  {
358  lstColors.Items[tempIndex] = prev;
359  _colorSource.SetColor(tempIndex, _colorId, prev);
360 
361  CurrentColorChanged?.Invoke(this, EventArgs.Empty);
362  }
363  }
364 
365  _dlgColor.OnColorChanged -= _dlgColor_OnColorChanged;
366  tempIndex = -1;
367  }
368  else
369  {
370  //Sort indices
371  int[] sorted = new int[count];
372  indices.CopyTo(sorted, 0);
373  Array.Sort(sorted);
374 
375  _dlgGradient.StartColor = (Color) (ARGBPixel) lstColors.Items[sorted[0]];
376  _dlgGradient.EndColor = (Color) (ARGBPixel) lstColors.Items[sorted[count - 1]];
377  if (_dlgGradient.ShowDialog(this) == DialogResult.OK)
378  {
379  //Interpolate and apply to each in succession.
380  ARGBPixel start = (ARGBPixel) _dlgGradient.StartColor;
381  ARGBPixel end = (ARGBPixel) _dlgGradient.EndColor;
382  float stepA = (end.A - start.A) / (float) count;
383  float stepR = (end.R - start.R) / (float) count;
384  float stepG = (end.G - start.G) / (float) count;
385  float stepB = (end.B - start.B) / (float) count;
386  for (int i = 0; i < count; i++)
387  {
388  ARGBPixel p = new ARGBPixel(
389  (byte) (start.A + i * stepA),
390  (byte) (start.R + i * stepR),
391  (byte) (start.G + i * stepG),
392  (byte) (start.B + i * stepB));
393  lstColors.Items[sorted[i]] = p;
394  _colorSource.SetColor(sorted[i], _colorId, p);
395  }
396  }
397  }
