559  public static void Build(ResourceGroup* group, int index, VoidPtr dataAddress, BRESString* pString)
560  {
561  //Get the first entry in the group, which is empty
562  ResourceEntry* list = &group->_first;
563  //Get the entry that will be modified
564  ResourceEntry* entry = &list[index];
565  //Get the first entry again
566  ResourceEntry* prev = &list[0];
567  //Get the entry that the first entry's left index points to
568  ResourceEntry* current = &list[prev->_leftIndex];
569  //The index of the current entry
570  ushort currentIndex = prev->_leftIndex;
571 
572  bool isRight = false;
573 
574  //Get the length of the string
575  int strLen = pString->_length;
576  if (strLen < 0 || strLen > ushort.MaxValue)
577  {
578  return;
579  }
580 
581  //Create a byte pointer to the struct's string data
582  byte* pChar = (byte*) pString + 4, sChar;
583 
584  int eIndex = strLen - 1, eBits = pChar[eIndex].CompareBits(0), val;
585  *entry = new ResourceEntry((eIndex << 3) | eBits, index, index, (int) dataAddress - (int) group,
586  (int) pChar - (int) group);
587 
588  //Continue while the previous id is greater than the current. Loop backs will stop the processing.
589  //Continue while the entry id is less than or equal the current id. Being higher than the current id means we've found a place to insert.
590  while (entry->_id <= current->_id && prev->_id > current->_id)
591  {
592  if (entry->_id == current->_id)
593  {
594  sChar = (byte*) group + current->_stringOffset;
595 
596  //Rebuild new id relative to current entry
597  for (eIndex = strLen; eIndex-- > 0 && pChar[eIndex] == sChar[eIndex];)
598  {
599  ;
600  }
601 
602  eBits = pChar[eIndex].CompareBits(sChar[eIndex]);
603 
604  entry->_id = (ushort) ((eIndex << 3) | eBits);
605 
606  if (((sChar[eIndex] >> eBits) & 1) != 0)
607  {
608  entry->_leftIndex = (ushort) index;
609  entry->_rightIndex = currentIndex;
610  }
611  else
612  {
613  entry->_leftIndex = currentIndex;
614  entry->_rightIndex = (ushort) index;
615  }
616  }
617 
618  //Is entry to the right or left of current?
619  isRight = (val = current->_id >> 3) < strLen && ((pChar[val] >> (current->_id & 7)) & 1) != 0;
620 
621  prev = current;
622  current = &list[currentIndex = isRight ? current->_rightIndex : current->_leftIndex];
623  }
624 
625  sChar = current->_stringOffset == 0 ? null : (byte*) group + current->_stringOffset;
626  val = sChar == null ? 0 : (int) *(bint*) (sChar - 4);
627 
628  if (val == strLen && ((sChar[eIndex] >> eBits) & 1) != 0)
629  {
630  entry->_rightIndex = currentIndex;
631  }
632  else
633  {
634  entry->_leftIndex = currentIndex;
635  }
636 
637  if (isRight)
638  {
639  prev->_rightIndex = (ushort) index;
640  }
641  else
642  {
643  prev->_leftIndex = (ushort) index;
644  }
645  }
