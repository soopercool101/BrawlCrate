402  public MDL0ShaderNode NewShader()
403  {
404  MDL0Node model = (MDL0Node) _resource;
405 
406  if (model._shadGroup == null)
407  {
408  MDL0GroupNode g = model._shadGroup;
409  if (g == null)
410  {
411  model.AddChild(g = new MDL0GroupNode(MDLResourceType.Shaders), true);
412  model._shadGroup = g;
413  model._shadList = g.Children;
414  }
415  }
416 
417  if (model._shadList != null &&
418  model._matList != null &&
419  model._shadList.Count < model._matList.Count)
420  {
421  MDL0ShaderNode shader = new MDL0ShaderNode();
422  model._shadGroup.AddChild(shader);
423  shader.Default();
424  shader.Rebuild(true);
425 
426  BaseWrapper b = FindResource(shader, true);
427  b?.EnsureVisible();
428 
429  return shader;
430  }
431 
432  MessageBox.Show("Shader could not be added. Make sure that you do not have more shaders than materials",
433  "Error");
434  return null;
435  }
