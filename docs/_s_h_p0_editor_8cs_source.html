623  {
624  //Set vertices (0), normals (1), and/or colors (2, 3)
625  //UVs are not morphed so there's no need to set them
626 
627  if (SelectedAnimation == null ||
628  SelectedSource == null || TargetModel?.Objects == null || TargetModel.Objects.Length == 0)
629  {
630  return;
631  }
632 
633  SHP0EntryNode entry = SelectedAnimation.FindChild(SelectedSource, false) as SHP0EntryNode;
634  if (entry == null)
635  {
636  return;
637  }
638 
639  if (MessageBox.Show(this,

641  "Are you sure?", MessageBoxButtons.OKCancel) == DialogResult.Cancel)
642  {
643  return;
644  }
645 
646  //Set the model to be only the bind pose with the SHP0 applied
647  //This is so when the data is unweighted,

652  CHR0Node chr = _mainWindow.SelectedCHR0;

660  foreach (ResourceNode n in nodes)

664  MDL0VertexNode node = (MDL0VertexNode) n;
665  MDL0ObjectNode[] o = new MDL0ObjectNode[node._objects.Count];
