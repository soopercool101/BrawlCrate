490  public void GenerateImports()
491  {
492  _imports.Clear();
493  Dictionary<uint, ImportData> tempImports = new Dictionary<uint, ImportData>();
494  foreach (ModuleSectionNode s in _sections)
495  {
496  foreach (ImportData e in tempImports.Values)
497  {
498  e._newSection = true;
499  e._lastOffset = 0;
500  }
501 
502  uint offset = 0;
503  List<RELLink> links;
504 
505  //Iterate through each command in the section
506  KeyValuePair<int, RelCommand>[] commands = s._manager.GetCommands();
507  foreach (KeyValuePair<int, RelCommand> r in commands)
508  {
509  RelCommand command = r.Value;
510  int index = r.Key;
511 
512  ImportData impData;
513  uint moduleID = command._moduleID;
514 
515  //Check if an import has been created for the target module.
516  if (_imports.ContainsKey(moduleID))
517  {
518  //An import already exists, so we'll add to it.
519  links = _imports[moduleID];
520  impData = tempImports[moduleID];
521  }
522  else
523  {
524  //An import does not exist, so it must be made.
525  _imports.Add(moduleID, links = new List<RELLink>());
526 
527  //Create new temporary import data
528  tempImports.Add(moduleID, impData = new ImportData {_newSection = true, _lastOffset = 0});
529  }
530 
531  //This is true when a new section is being evaluated.
532  if (impData._newSection)
533  {
534  links.Add(new RELLink {_type = RELLinkType.Section, _section = (byte) s.Index});
535  impData._newSection = false;
536  }
537 
538  //Get the offset of the command within the section.
539  offset = (uint) index * 4 + (command.IsHalf ? 2u : 0);
540 
541  //Get the offset to this address relative to the last written link offset.
542  uint diff = offset - impData._lastOffset;
543 
544  //If the difference is greater than ushort allows,
545  //add increment links until the difference works
546  while (diff > 0xFFFF)
547  {
548  impData._lastOffset += 0xFFFF;
549  diff = offset - impData._lastOffset;
550 
551  links.Add(new RELLink
552  {_type = RELLinkType.IncrementOffset, _section = 0, _value = 0, _prevOffset = 0xFFFF});
553  }
554 
555  //Gather the link information
556  byte targetSection = (byte) command._targetSectionId;
557  RELLinkType type = (RELLinkType) command._command;
558  uint val = command._addend;
559 
560  //Write command link
561  links.Add(new RELLink
562  {_type = type, _section = targetSection, _value = val, _prevOffset = (ushort) diff});
563 
564  //Don't bother adding the difference,
565  //just set the exact offset as the last offset
566  impData._lastOffset = offset;
567  }
568  }
569 
570  foreach (List<RELLink> cmds in _imports.Values)
571  {
572  cmds.Add(new RELLink {_type = RELLinkType.End});
573  }
