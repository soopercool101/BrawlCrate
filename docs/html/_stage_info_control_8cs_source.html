281  private static int stageIDScan(FileInfo relFile, bool fix)
282  {
283  FileStream stream;
284  stream = relFile.Open(FileMode.Open, FileAccess.ReadWrite);
285 
286  // search through pointer
287  long length = relFile.Length;
288  byte[] searchFor = {0x38, 0xa5, 0x00, 0x00, 0x38, 0x80, 0x00};
289  int indexToCheck = 0;
290  bool found = false;
291 
292  int i = 0;
293  while (!found && i < length)
294  {
295  if (stream.ReadByte() == searchFor[indexToCheck])
296  {
297  indexToCheck++;
298  if (indexToCheck == searchFor.Length)
299  {
300  if (StageModuleConverter.IndicesToIgnore.Contains(i + 1))
301  {
302  //MessageBox.Show("ignored " + (i + 1));
303  indexToCheck = 0;
304  }
305  else
306  {
307  found = true;
308  if (fix)
309  {
310  stream.WriteByte((byte) getIdealStageID(relFile.Name));
311  }
312  }
313  }
314  }
315  else
316  {
317  indexToCheck = 0;
318  }
319 
320  i++;
321  }
322 
323  int b = stream.ReadByte();
324  stream.Close();
325  return b;
326  }
