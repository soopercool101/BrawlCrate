385  public virtual ResourceNode Duplicate(bool changeName)
386  {
387  if (_resource.Parent == null)
388  {
389  return null;
390  }
391 
392  string tempPath = Path.GetTempFileName();
393  _resource.Export(tempPath);
394  // Initialize node as a child of the parent
395  ResourceNode rNode2 = NodeFactory.FromFile(_resource.Parent, tempPath, _resource.GetType());
396 
397  if (rNode2 == null)
398  {
399  MessageBox.Show("The node could not be duplicated correctly.");
400  return null;
401  }
402 
403  // Remove the node from the parent temporarily
404  rNode2.Remove();
405 
406  // Copy ARCEntryNode data, which is contained in the containing ARC, not the node itself
407  if (rNode2 is ARCEntryNode)
408  {
409  ((ARCEntryNode) rNode2).FileIndex = ((ARCEntryNode) _resource).FileIndex;
410  ((ARCEntryNode) rNode2).FileType = ((ARCEntryNode) _resource).FileType;
411  ((ARCEntryNode) rNode2).GroupID = ((ARCEntryNode) _resource).GroupID;
412  ((ARCEntryNode) rNode2).RedirectIndex = ((ARCEntryNode) _resource).RedirectIndex;
413  }
414 
415  // Copy the name directly in cases where name isn't saved
416  rNode2.Name = _resource.Name;
417  // Set the name programatically (based on Windows' implementation)
418  int index = _resource.Index;
419  int n = 0;
420  if (changeName)
421  {
422  while (_resource.Parent.FindChildrenByName(rNode2.Name).Length >= 1)
423  {
424  // Get the last index of the last duplicated node in order to place it after that one
425  index = Math.Max(index, _resource.Parent.FindChildrenByName(rNode2.Name).Last().Index);
426  // Set the name based on the number of duplicate nodes found
427  rNode2.Name = $"{_resource.Name} ({++n})";
428  }
429  }
430 
431  // Place the node in the same containing parent, after the last duplicated node.
432  _resource.Parent.InsertChild(rNode2, true, index + 1);
433 
434  return rNode2;
435  }
