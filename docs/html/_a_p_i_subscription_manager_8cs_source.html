445  public static void GetNewFiles()
446  {
447  //DepreciatedReplacementStrings.Keys.Any(s => e.Message.Contains(s))
448  List<string> newLoaders = new List<string>();
449  List<string> dangerousFiles = new List<string>();
450  List<FileInfo> newFiles = new List<FileInfo>();
451  foreach (FileInfo f in Directory.CreateDirectory(Program.ApiPath).GetFiles("*.new"))
452  {
453  newLoaders.AddRange(File.ReadAllLines(f.FullName).Where(o => o.StartsWith("Loaders/")));
454  newFiles.Add(f);
455  }
456 
457  while (newFiles.Count > 0)
458  {
459  try
460  {
461  newFiles[0].Delete();
462  }
463  catch
464  {
465  // ignored. It should hopefully not happen, but disabling new loaders is more important.
466  }
467 
468  newFiles.RemoveAt(0);
469  }
470 
471  if (newLoaders.Count == 0)
472  {
473  return;
474  }
475 
476 
477  // Ensure that the format properly matches that used by the lists
478  for (int i = 0; i < newLoaders.Count; i++)
479  {
480  newLoaders[i] = newLoaders[i].Replace('/', '\\');
481  if (newLoaders[i].StartsWith("Loaders\\", StringComparison.OrdinalIgnoreCase))
482  {
483  newLoaders[i] = newLoaders[i].Substring(8);
484  }
485  }
486 
487  string message =
488  "The following loaders that are currently active have been added or changed by updates: \n\n";
489  List<string> loadersToDisable = new List<string>();
490  if (Properties.Settings.Default.APIOnlyAllowLoadersFromWhitelist)
491  {
492  foreach (string s in Properties.Settings.Default.APILoadersWhitelist)
493  {
494  if (newLoaders.Any(o => o.Equals(s, StringComparison.OrdinalIgnoreCase)))
495  {
496  message += s + '\n';
497  loadersToDisable.Add(s);
498  }
499  }
500  }
501  else
502  {
503  foreach (string s in newLoaders)
504  {
505  if (!Properties.Settings.Default.APILoadersBlacklist.Cast<string>().ToArray()
506  .Any(o => o.Equals(s, StringComparison.OrdinalIgnoreCase)))
507  {
508  message += s + '\n';
509  loadersToDisable.Add(s);
510  }
511  }
512  }
513 
514  if (loadersToDisable.Count > 0)
515  {
516  message += "\nLoaders downloaded from the internet may be harmful, so these have been deactivated. " +
517  "It is recommended that you review these files before enabling them again. " +
518  "Would you like to enable these files?\n\n" +
519  "The BrawlCrate team accepts no responsibility for any files that may harm your computer.";
520  if (MessageBox.Show(message, "Warning", MessageBoxButtons.YesNo, MessageBoxIcon.Warning) !=
521  DialogResult.Yes)
522  {
523  // Deactivate them
524  // Use whitelist if that is what is used currently:
525  if (Properties.Settings.Default.APIOnlyAllowLoadersFromWhitelist)
526  {
527  StringCollection sc = new StringCollection();
528  foreach (string s in Properties.Settings.Default.APILoadersWhitelist)
529  {
530  if (!loadersToDisable.Any(o => o.Equals(s, StringComparison.OrdinalIgnoreCase)))
531  {
532  sc.Add(s);
533  }
534  }
535 
536  Properties.Settings.Default.APILoadersWhitelist = sc;
537  Properties.Settings.Default.Save();
538  }
539  else // Add them to the blacklist
540  {
541  StringCollection sc = Properties.Settings.Default.APILoadersBlacklist;
542  sc.AddRange(loadersToDisable.ToArray());
543 
544  Properties.Settings.Default.APILoadersBlacklist = sc;
545  Properties.Settings.Default.Save();
546  }
547  }
548  }

552  public class APISubscription : ListViewItem
553  {
554  public string Version { get; }
555  public string LastUpdateDate { get; }
556  public List<string> ManagedScripts { get; }
557  public string ReadMe { get; }
558  public string License { get; }
559 
560  public APISubscription(string repoOwner, string repoName, string path)
561  {
562  Name = $"{repoOwner}/{repoName}";
563  ReadMe = "";
564  License = "";
565  if (File.Exists($"{path} README.txt"))
566  {
567  ReadMe = File.ReadAllText($"{path} README.txt");
568  }
569 
570  if (File.Exists($"{path} LICENSE.txt"))
571  {
572  License = File.ReadAllText($"{path} LICENSE.txt");
573  }
574 
575  string[] lines = File.ReadAllLines(path);
576  // Line 0: Release Tag. This is checked against to see if there is a new update for the repo.
577  // Line 1: Release Target Commitish. Used to allow continuous integration repos to work.
578  // Line 2: Update date (not used by updater, used instead to view info)
579  // Line 3+: Each line is a relative path to a file from the installation.
580  // This is used to delete relevant files when updating,
581  // in case a file is moved or deleted by the update intentionally.
582  Version = lines[0];
583  LastUpdateDate = lines[2];
584  ManagedScripts = new List<string>();
585  for (int i = 3; i < lines.Length; i++)
586  {
587  ManagedScripts.Add(lines[i]);
588  }
589 
590  Text = ToString();
591  }
592 
593  public override string ToString()
594  {
595  return $"{Name} ({Version})";
596  }
597  }
