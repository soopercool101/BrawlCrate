240  public static string ApplyExtension(string path, string filter, int filterIndex)
241  {
242  int tmp;
243  if (Path.HasExtension(path) && !int.TryParse(Path.GetExtension(path), out tmp))
244  {
245  return path;
246  }
247 
248  int index = filter.IndexOfOccurance('|', filterIndex * 2);
249  if (index == -1)

252  }
253

255  int len = Math.Max(filter.Length, filter.IndexOfAny(new char[] {';', '|'})) - index;
256
