152  private static CMPRBlock optimalCompressDXT1(ARGBPixel c)
153  {
154  CMPRBlock block = new CMPRBlock();
155 
156  uint indices = 0xAAAAAAAA;
157 
158  block._root0._data = (ushort) ((OMatch5[c.R, 0] << 11) | (OMatch6[c.G, 0] << 5) | OMatch5[c.B, 0]);
159  block._root1._data = (ushort) ((OMatch5[c.R, 1] << 11) | (OMatch6[c.G, 1] << 5) | OMatch5[c.B, 1]);
160 
161  if (block._root0 < block._root1)
162  {
163  VoidPtr.Swap((short*) &block._root0, (short*) &block._root1);
164  indices ^= 0x55555555;
165  }
166 
167  block._lookup = indices;
168  return block;
169  }

224  private static void selectDiagonal(Vector3* points, int num, Vector3* maxColor, Vector3* minColor)
225  {
226  Vector3 center = (*maxColor + *minColor) * 0.5f;
227  Vector3 t;
228  Vector2* tp = (Vector2*) &t;
229 
230  Vector2 covariance = new Vector2();
231  for (uint i = 0; i < num; i++)
232  {
233  t = points[i] - center;
234  covariance += *tp * t._z;
235  }
236 
237  float x0 = maxColor->_x;
238  float y0 = maxColor->_y;
239  float x1 = minColor->_x;
240  float y1 = minColor->_y;
241 
242  if (covariance._x < 0)
243  {
244  VoidPtr.Swap(&x0, &x1);
245  }
246 
247  if (covariance._y < 0)
248  {
249  VoidPtr.Swap(&y0, &y1);
250  }
251 
252  maxColor->_x = x0;
253  maxColor->_y = y0;
254 
255  minColor->_x = x1;
256  minColor->_y = y1;

270  private static ushort roundAndExpand(Vector3* v)
271  {
272  uint r = (uint) (Math.Min(Math.Max(v->_x * (31.0f / 255.0f), 0.0f), 31.0f) + 0.5f);
273  uint g = (uint) (Math.Min(Math.Max(v->_y * (63.0f / 255.0f), 0.0f), 63.0f) + 0.5f);
274  uint b = (uint) (Math.Min(Math.Max(v->_z * (31.0f / 255.0f), 0.0f), 31.0f) + 0.5f);
275 
276 
277  ushort w = (ushort) ((r << 11) | (g << 5) | b);
278 
279  r = (r << 3) | (r >> 2);
280  g = (g << 2) | (g >> 4);
281  b = (b << 3) | (b >> 2);
282  *v = new Vector3(r, g, b);
283 
284  return w;
285  }

383  private static void optimizeEndPoints4(Vector3* points, CMPRBlock* block)
384  {
385  float alpha2_sum = 0.0f;
386  float beta2_sum = 0.0f;
387  float alphabeta_sum = 0.0f;
388  Vector3 alphax_sum = new Vector3();
389  Vector3 betax_sum = new Vector3();
390  uint indices = block->_lookup;
391 
392  for (int i = 0, bi = 30; i < 16; ++i, bi -= 2)
393  {
394  uint bits = indices >> bi;
395 
396  float beta = bits & 1;
397  if ((bits & 2) != 0)
398  {
399  beta = (1 + beta) / 3.0f;
400  }
401 
402  float alpha = 1.0f - beta;
403 
404  alpha2_sum += alpha * alpha;
405  beta2_sum += beta * beta;
406  alphabeta_sum += alpha * beta;
407  alphax_sum += alpha * points[i];
408  betax_sum += beta * points[i];
409  }
410 
411  float denom = alpha2_sum * beta2_sum - alphabeta_sum * alphabeta_sum;
412  if (Math.Abs(denom) <= 0.0001f)
413  {
414  return;
415  }
416 
417  float factor = 1.0f / denom;
418 
419  Vector3 a = (alphax_sum * beta2_sum - betax_sum * alphabeta_sum) * factor;
420  Vector3 b = (betax_sum * alpha2_sum - alphax_sum * alphabeta_sum) * factor;
421 
422  a.Clamp(0.0f, 255.0f);
423  b.Clamp(0.0f, 255.0f);
424 
425  ushort color0 = roundAndExpand(&a);
426  ushort color1 = roundAndExpand(&b);
427 
428  if (color0 < color1)
429  {
430  Vector3 t = a;
431  a = b;
432  b = t;
433  VoidPtr.Swap(&color0, &color1);
434  }
435 
436  //CMPRBlock block = new CMPRBlock();
437  block->_root0._data = color0;
438  block->_root1._data = color1;
439  block->_lookup = computeIndices4(points, &a, &b);
440  }
