270  private void RaiseUpdateFrame()
271  {
272  int numUpdates = 0;
273  double totalUpdateTime = 0;
274 
275  // Cap the maximum time drift to 1 second (e.g. when the process is suspended).
276  double time = _updateWatch.Elapsed.TotalSeconds;
277  if (time <= 0)
278  {
279  return;
280  }
281 
282  if (time > 1.0)
283  {
284  time = 1.0;
285  }
286 
287  // Raise UpdateFrame events until we catch up with our target update rate.
288  while (_nextUpdate - time <= 0 && time > 0)
289  {
290  _nextUpdate -= time;
291  OnUpdateFrameInternal(new FrameEventArgs(time));
292  time = _updateTime = _updateWatch.Elapsed.TotalSeconds - time;
293  // Stopwatches are not accurate over long time periods.
294  // We accumulate the total elapsed time into the time variable
295  // while reseting the Stopwatch frequently.
296  _updateWatch.Reset();
297  _updateWatch.Start();
298 
299  // Don't schedule a new update more than 1 second in the future.
300  // Sometimes the hardware cannot keep up with updates
301  // (e.g. when the update rate is too high, or the UpdateFrame processing
302  // is too costly). This cap ensures we can catch up in a reasonable time
303  // once the load becomes lighter.
304  _nextUpdate += TargetUpdatePeriod;
305  _nextUpdate = Math.Max(_nextUpdate, -1.0);
306 
307  totalUpdateTime += _updateTime;
308 
309  // Allow up to 10 consecutive UpdateFrame events to prevent the
310  // application from "hanging" when the hardware cannot keep up
311  // with the requested update rate.
312  if (++numUpdates >= 10 || TargetUpdateFrequency == 0.0)
313  {
314  break;
315  }
316  }
317 
318  // Calculate statistics
319  if (numUpdates > 0)
320  {
321  _updatePeriod = totalUpdateTime / numUpdates;
322  }
