842  {
843  StringBuilder sb = new StringBuilder();
844  using (FileStream stream = new FileStream(f.FullName, FileMode.Open, FileAccess.Read))
845  {
846  stream.Seek(16, SeekOrigin.Begin);
847  int b = stream.ReadByte();
848  while (b == 0)
849  {
850  b = stream.ReadByte();
851  }
852 
853  while (b != 0)
854  {
855  sb.Append((char) b);
856  b = stream.ReadByte();
857  }
858  }
859 
860  if (sb.ToString().IndexOfAny(Path.GetInvalidFileNameChars()) > -1)
861  {
862  return f.Name;
863  }
864 
865  return sb + ".pac";
