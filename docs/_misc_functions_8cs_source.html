574  private bool PointCollides(Vector3 point, out float y_result)
575  {
576  y_result = float.MaxValue;
577  Vector2 v2 = new Vector2(point._x, point._y);
578  foreach (CollisionNode coll in _collisions)
579  {
580  foreach (CollisionObject obj in coll.Children)
581  {
582  if (obj._render)
583  {
584  foreach (CollisionPlane plane in obj._planes)
585  {
586  if (plane._type == CollisionPlaneType.Floor && plane.IsCharacters)
587  {
588  if (plane.PointLeft._x <= v2._x && plane.PointRight._x >= v2._x)
589  {
590  float x = v2._x;
591  float m = (plane.PointLeft._y - plane.PointRight._y)
592  / (plane.PointLeft._x - plane.PointRight._x);
593  float b = plane.PointRight._y - m * plane.PointRight._x;
594  float y_target = m * x + b;
595  //Console.WriteLine(y_target);
596  if (Math.Abs(y_target - v2._y) <= Math.Abs(y_result - v2._y))
597  {
598  y_result = y_target;
599  }
600  }
601  }
602  }
603  }
604  }
605  }
606 
607  return Math.Abs(y_result - v2._y) <= 5;
608  }
