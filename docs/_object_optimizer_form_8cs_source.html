84  TriangleConverter triConverter = new TriangleConverter(chkUseStrips.Checked, (uint) numCacheSize.Value,
85  (uint) numMinStripLen.Value, chkPushCacheHits.Checked);
86  Facepoint[] points = new Facepoint[a._object._manager._triangles._indices.Length];
87  uint[] indices = a._object._manager._triangles._indices;
88  bool ccw = Collada._importOptions._forceCCW;
89 
90  //Indices are written in reverse for each triangle,

92  for (int t = 0; t < a._object._manager._triangles._indices.Length; t++)
93  {
94  points[ccw ? t : t - t % 3 + (2 - t % 3)] = a._facepoints[indices[t]];
95  }
96 
97  List<PrimitiveGroup> p = triConverter.GroupPrimitives(points, out int pc, out int fc);
98

103  a._groups = p;
104  }
