77  // Algorithm: use open addressing double hashing (no chaining) on the
78  // prefix code / next character combination. We do a variant of Knuth's
79  // algorithm D (vol. 3, sec. 6.4) along with G. Knott's relatively-prime
80  // secondary probe. Here, the modular division first probe is gives way
81  // to a faster exclusive-or manipulation. Also do block compression with
82  // an adaptive reset, whereby the code table is cleared when the compression
83  // ratio decreases, but after the table fills. The variable-length output
84  // codes are re-sized at this point, and a special CLEAR code is generated
85  // for the decompressor. Late addition: construct the table according to
86  // file size for noticeable speed improvement on small files. Please direct
87  // questions about this implementation to ames!jaw.

179  private void Compress(int init_bits, Stream outs)
180  {
181  int fcode;
182  int i /* = 0 */;
183  int c;
184  int ent;
185  int disp;
186  int hsize_reg;
187  int hshift;
188 
189  // Set up the globals: g_init_bits - initial number of bits
190  g_init_bits = init_bits;
191 
192  // Set up the necessary values
193  clear_flg = false;
194  n_bits = g_init_bits;
195  maxcode = MaxCode(n_bits);
196 
197  ClearCode = 1 << (init_bits - 1);
198  EOFCode = ClearCode + 1;
199  free_ent = ClearCode + 2;
200 
201  a_count = 0; // clear packet
202 
203  ent = NextPixel();
204 
205  hshift = 0;
206  for (fcode = hsize; fcode < 65536; fcode *= 2)
207  {
208  ++hshift;
209  }
210 
211  hshift = 8 - hshift; // set hash code range bound
212 
213  hsize_reg = hsize;
214  ResetCodeTable(hsize_reg); // clear hash table
215 
216  Output(ClearCode, outs);
217 
218  outer_loop:
219  while ((c = NextPixel()) != EOF)
220  {
221  fcode = (c << maxbits) + ent;
222  i = (c << hshift) ^ ent; // xor hashing
223 
224  if (htab[i] == fcode)
225  {
226  ent = codetab[i];
227  continue;
228  }
229  else if (htab[i] >= 0) // non-empty slot
230  {
231  disp = hsize_reg - i; // secondary hash (after G. Knott)
232  if (i == 0)
233  {
234  disp = 1;
235  }
236 
237  do
238  {
239  if ((i -= disp) < 0)
240  {
241  i += hsize_reg;
242  }
243 
244  if (htab[i] == fcode)
245  {
246  ent = codetab[i];
247  goto outer_loop;
248  }
249  } while (htab[i] >= 0);
250  }
251 
252  Output(ent, outs);
253  ent = c;
254  if (free_ent < maxmaxcode)
255  {
256  codetab[i] = free_ent++; // code -> hashtable
257  htab[i] = fcode;
258  }
259  else
260  {
261  ClearTable(outs);
262  }
263  }
264 
265  // Put out the final code.
266  Output(ent, outs);
267  Output(EOFCode, outs);
268  }

322  private void Output(int code, Stream outs)
323  {
324  cur_accum &= masks[cur_bits];
325 
326  if (cur_bits > 0)
327  {
328  cur_accum |= code << cur_bits;
329  }
330  else
331  {
332  cur_accum = code;
333  }
334 
335  cur_bits += n_bits;
336 
337  while (cur_bits >= 8)
338  {
339  Add((byte) (cur_accum & 0xff), outs);
340  cur_accum >>= 8;
341  cur_bits -= 8;
342  }
343 
344  // If the next entry is going to be too big for the code size,
345  // then increase it, if possible.
346  if (free_ent > maxcode || clear_flg)
347  {
348  if (clear_flg)
349  {
350  maxcode = MaxCode(n_bits = g_init_bits);
351  clear_flg = false;
352  }
353  else
354  {
355  ++n_bits;
356  if (n_bits == maxbits)
357  {
358  maxcode = maxmaxcode;
359  }
360  else
361  {
362  maxcode = MaxCode(n_bits);
363  }
364  }
365  }
366 
367  if (code == EOFCode)
368  {
369  // At EOF, write the rest of the buffer.
370  while (cur_bits > 0)
371  {
372  Add((byte) (cur_accum & 0xff), outs);
373  cur_accum >>= 8;
374  cur_bits -= 8;
375  }
376 
377  Flush(outs);
378  }
