574  //Returns true if the vertex's matrix node is changed.
575  private bool Weight(float value, Vertex3 vertex, WeightType type)
576  {
577  if (_targetBone == null || _targetBone.Locked)
578  {
579  return false;
580  }
581 
582  Influence targetInf = null;
583  BoneWeight targetWeight = null;
584  float max = 1.0f;
585 
586  //Get the matrix that influences this vertex
587  IMatrixNode node = vertex.GetMatrixNode();
588 
589  bool startsAsBone = false;
590 
591  //Convert a single bone into an influence so bones can be added to it later
592  if (node is MDL0BoneNode)
593  {
594  startsAsBone = true;
595  node = new Influence(node as MDL0BoneNode);
596  }
597 
598  //Duplicate the influence if it affects more than just this vertex
599  targetInf = node.Users.Count > 1 ? (node as Influence).Clone() : node as Influence;
600 
601  //Find or add the current bone to the influence
602  List<BoneWeight> weights = targetInf.Weights;
603  int selectedIndex = weights.Select(x => x.Bone).ToArray().IndexOf(TargetBone);
604  if (selectedIndex < 0)
605  {
606  weights.Add(new BoneWeight(TargetBone, 0.0f));
607  selectedIndex = weights.Count - 1;
608  }
609 
610  //Get the weight at the index of the current bone
611  targetWeight = targetInf.Weights[selectedIndex];
612 
613  //Can't do anything to a locked weight
614  if (targetWeight.Locked)
615  {
616  return false;
617  }
618 
619  //Get the sum of all weights that can be edited by subtracting all locked values from 1.0f
620  max = 1.0f;
621  foreach (BoneWeight b in weights)
622  {
623  if (b.Locked)
624  {
625  max -= b.Weight;
626  }
627  }
628 
629  //Get the new value for the target weight
630  //Clamp it between 0.0f and the max value
631  switch (type)
632  {
633  default:
634  value = RoundValue(value, max);
635  break;
636  case WeightType.Add:
637  value = RoundValue(targetWeight.Weight + value, max);
638  break;
639  case WeightType.Multiply:
640  value = RoundValue(targetWeight.Weight * value, max);
641  break;
642  }
643 
644  //Nothing to do if there's no change in value
645  if (targetWeight.Weight == value)
646  {
647  return false;
648  }
649 
650  //Collect all unlocked weights that are not the current weight
651  //These are weights that will be changed to accomodate the current weight edit
652  List<int> editableWeights = new List<int>();
653  for (int i = 0; i < targetInf.Weights.Count; i++)
654  {
655  if (!targetInf.Weights[i].Locked && i != selectedIndex)
656  {
657  editableWeights.Add(i);
658  }
659  }
660 
661  //Return if nothing can be edited
662  if (editableWeights.Count == 0)
663  {
664  return false;
665  }
666 
667  //Set the current weight with the calculated value
668  targetWeight.Weight = value;
669 
670  //Get the change in value, divide it by all other editable weights,
671  //and then add that value to those weights to bring the overall weight sum back to 1.0f
672  float perBoneDiff = (targetWeight.Weight - value) / editableWeights.Count;
673  if (value < max)
674  {
675  foreach (int i in editableWeights)
676  {
677  targetInf.Weights[i].Weight = RoundValue(targetInf.Weights[i].Weight + perBoneDiff, 1.0f);
678  }
679  }
680  else
681  {
682  foreach (int i in editableWeights)
683  {
684  targetInf.Weights[i].Weight = 0.0f;
685  }
686  }
687 
688  //Normalize the influence just in case, this will scale all weights so they add up to 1.0f
689  //Don't let the modified value be normalized, lock it
690  bool locked = targetWeight.Locked;
691  targetWeight.Locked = true;
692  targetInf.Normalize();
693  targetWeight.Locked = locked;
694 
695  //Clean influence by removing zero weights
696  for (int i = 0; i < targetInf.Weights.Count; i++)
697  {
698  if (targetInf.Weights[i].Weight <= 0.0f)
699  {
700  targetInf.Weights.RemoveAt(i--);
701  }
702  }
703 
704  MDL0ObjectNode obj = vertex.Parent as MDL0ObjectNode;
705  MDL0Node model = obj.Model;
706  IMatrixNode matrixNode;
707 
708  //See if the influence is just one bone
709  if (targetInf.Weights.Count == 1)
710  {
711  matrixNode = targetInf.Weights[0].Bone;
712  if (!startsAsBone && !_anyConverted.Contains(obj))
713  {
714  _anyConverted.Add(obj);
715  }
716  }
717  else
718  {
719  matrixNode = model._influences.FindOrCreate(targetInf);
720  if (startsAsBone && !_anyConverted.Contains(obj))
721  {
722  _anyConverted.Add(obj);
723  }
724  }
725 
726  //Move influence to each vertex before modifying the influence of one vertex
727  if (obj.MatrixNode != null)
728  {
729  obj.TryConvertMatrixToVertex();
730  }
731 
732  vertex.MatrixNode = matrixNode;
733 
734  if (obj.MatrixNode == null)
735  {
736  obj.TryConvertMatrixToObject();
737  }
738 
739  return true;
740  }
