465  if (suffix == null)

475  // Initialize GIF data stream decoder.
476 
477  data_size = Read();
478  clear = 1 << data_size;
479  end_of_information = clear + 1;
480  available = clear + 2;
481  old_code = NullCode;
482  code_size = data_size + 1;
483  code_mask = (1 << code_size) - 1;
484  for (code = 0; code < clear; code++)
485  {
486  prefix[code] = 0;
487  suffix[code] = (byte) code;
488  }
489 
490  // Decode GIF pixel stream.
491 
492  datum = bits = count = first = top = pi = bi = 0;
493 
494  for (i = 0; i < npix;)
495  {
496  if (top == 0)
497  {
498  if (bits < code_size)
499  {
500  // Load bytes until there are enough bits for a code.
501  if (count == 0)
502  {
503  // Read a new data block.
504  count = ReadBlock();
505  if (count <= 0)
506  {
507  break;
508  }
509 
510  bi = 0;
511  }
512 
513  datum += (block[bi] & 0xff) << bits;
514  bits += 8;
515  bi++;
516  count--;
517  continue;
518  }
519 
520  // Get the next code.
521 
522  code = datum & code_mask;
523  datum >>= code_size;
524  bits -= code_size;
525 
526  // Interpret the code
527 
528  if (code > available || code == end_of_information)
529  {
530  break;
531  }
532 
533  if (code == clear)
534  {
535  // Reset decoder.
536  code_size = data_size + 1;
537  code_mask = (1 << code_size) - 1;
538  available = clear + 2;
539  old_code = NullCode;
540  continue;
541  }
542 
543  if (old_code == NullCode)
544  {
545  pixelStack[top++] = suffix[code];
546  old_code = code;
547  first = code;
548  continue;
549  }
550 
551  in_code = code;
552  if (code == available)
553  {
554  pixelStack[top++] = (byte) first;
555  code = old_code;
556  }
557 
558  while (code > clear)
559  {
560  pixelStack[top++] = suffix[code];
561  code = prefix[code];
562  }
563 
564  first = suffix[code] & 0xff;
565 
566  // Add a new string to the string table,
567 
568  if (available >= MaxStackSize)
569  {
570  break;
571  }
572 
573  pixelStack[top++] = (byte) first;
574  prefix[available] = (short) old_code;
575  suffix[available] = (byte) first;
576  available++;
577  if ((available & code_mask) == 0
578  && available < MaxStackSize)
579  {
580  code_size++;
581  code_mask += available;
582  }
583 
584  old_code = in_code;
585  }
586

588 
589  top--;
590  pixels[pi++] = pixelStack[top];
591  i++;
592  }
593
