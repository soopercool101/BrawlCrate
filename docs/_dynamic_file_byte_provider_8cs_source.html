533  private void MoveFileBlock(FileDataBlock fileBlock, long dataOffset)
534  {
535  // First, determine whether the next file block needs to move before this one.
536  FileDataBlock nextFileBlock = GetNextFileDataBlock(fileBlock, dataOffset, out long nextDataOffset);
537  if (nextFileBlock != null && dataOffset + fileBlock.Length > nextFileBlock.FileOffset)
538  {
539  // The next block needs to move first, so do that now.
540  MoveFileBlock(nextFileBlock, nextDataOffset);
541  }
542 
543  // Now, move the block.
544  if (fileBlock.FileOffset > dataOffset)
545  {
546  // Move the section to earlier in the file stream (done in chunks starting at the beginning of the section).
547  byte[] buffer = new byte[COPY_BLOCK_SIZE];
548  for (long relativeOffset = 0; relativeOffset < fileBlock.Length; relativeOffset += buffer.Length)
549  {
550  long readOffset = fileBlock.FileOffset + relativeOffset;
551  int bytesToRead = (int) Math.Min(buffer.Length, fileBlock.Length - relativeOffset);
552  _stream.Position = readOffset;
553  _stream.Read(buffer, 0, bytesToRead);
554 
555  long writeOffset = dataOffset + relativeOffset;
556  _stream.Position = writeOffset;
557  _stream.Write(buffer, 0, bytesToRead);
558  }
559  }
560  else
561  {
562  // Move the section to later in the file stream (done in chunks starting at the end of the section).
563  byte[] buffer = new byte[COPY_BLOCK_SIZE];
564  for (long relativeOffset = 0; relativeOffset < fileBlock.Length; relativeOffset += buffer.Length)
565  {
566  int bytesToRead = (int) Math.Min(buffer.Length, fileBlock.Length - relativeOffset);
567  long readOffset = fileBlock.FileOffset + fileBlock.Length - relativeOffset - bytesToRead;
568  _stream.Position = readOffset;
